<script>

  const promo = document.querySelector(".store-promo");

  const observer = new IntersectionObserver(entries =>
  {
    entries.forEach(entry =>
    {
      if (entry.isIntersecting)
      {
        promo.classList.add("visible");
      }
    });
  },
  { threshold: .2 });

  observer.observe(promo);
